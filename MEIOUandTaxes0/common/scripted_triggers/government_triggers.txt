### Checking Republic
dictator_trigger = {
	#government = republic
	OR = {
		AND = {
			NOT = { has_reform = free_city }
			NOT = { has_reform = city_state }
			NOT = { has_reform = trade_republic }
			NOT = { has_reform = colonial_government }
			has_reform = dictator_reform
		}
		has_reform = republican_dictatorship
	}
}
