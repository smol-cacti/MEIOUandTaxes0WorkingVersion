			}
		}

		set_variable = { which = urban_migrants_tax value = 0 }
		set_variable = { which = urban_migrants_production value = 0 }
		set_variable = { which = urban_migrants_manpower value = 0 }
		set_variable = { which = refugee_wealth value = 0 }
		clr_country_flag = refugee_land_@FROM
		clr_country_flag = refugee_welcome_@ROOT
		every_owned_province = {
			clr_province_flag = refugee_target
			clr_province_flag = refugee_target_urban
			clr_province_flag = refugee_target_heretic
			clr_province_flag = refugee_target_undeveloped
			subtract_variable = { which = province_religion which = migrant_religion }
			set_variable = { which = original_population value = 0 }
			set_variable = { which = migrants value = 0 }
			set_variable = { which = migrant_population value = 0 }
			set_variable = { which = migrant_religion value = 0 }
		}
	}
