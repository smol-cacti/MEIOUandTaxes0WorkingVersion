AssessBuilding = {
	if = {
		limit = { can_build_$name$ = yes }
		AssessBuilding_$name$ = { root = $root$ weight = assessBuildingWeight }
		if = {
			limit = { check_variable = { which = assessBuildingWeight which = getBestBuildingWeight } }
			set_variable = { which = getBestBuildingWeight which = assessBuildingWeight }
			set_variable = { which = getBestBuildingBuildInd value = $ind$ }
			get_upgrade_cost_$name$ = yes
			multiply_variable = { which = building_cost which = getBestBuildingCost }
			set_variable = { which = getBestBuildingBuildPrice which = building_cost }
			set_variable = { which = building_cost value = 0 }
		}
		set_variable = { which = assessBuildingPrice value = 0 }
		set_variable = { which = assessBuildingWeight value = 0 }
	}
}
