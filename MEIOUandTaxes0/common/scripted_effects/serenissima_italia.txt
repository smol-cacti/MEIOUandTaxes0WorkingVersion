disable_italian_powers = {
	custom_tooltip = disable_italian_powers_tt
	hidden_effect = {
		set_global_flag = italian_powers_disabled
		AAA = {
			clr_country_flag = italian_powers_enabled
		}
		clr_country_flag = italian_power
		clr_country_flag = insignificant_italian_power
		clr_country_flag = local_italian_power
		clr_country_flag = regional_italian_power
		clr_country_flag = major_italian_power
		clr_country_flag = dominant_italian_power
		clr_country_flag = menacing_italian_power
		remove_country_modifier = insignificant_italian_power
		remove_country_modifier = local_italian_power
		remove_country_modifier = regional_italian_power
		remove_country_modifier = major_italian_power
		remove_country_modifier = dominant_italian_power
		remove_country_modifier = menacing_italian_power
		every_country = {
			limit = {
				has_country_flag = italian_power
			}
			clr_country_flag = italian_power
			country_event = { id = italian_power.5 }
		}
	}
}
assign_italian_power = { #40% #20% #10% #5% #2%
#Country dev outside of Italy reduces power by half of its dev
#Up to country dev outside of Italy won't reduce power
	multiply_variable = { which = temp value = 2 }
	if = {
		limit = {
			NOT = {
				check_variable = { which = temp value = 1 }
			}
		}
		multiply_variable = { which = italian_power which = temp }
	}
	if = {
		limit = {
			check_variable = { which = italian_power value = 0.1 }
		}
		if = {
			limit = {
				check_variable = { which = italian_power value = 0.2 }
			}
			if = {
				limit = {
					check_variable = { which = italian_power value = 0.4 }
				}
				assign_italian_power_helper = { power=menacing }
			}
			else = {
				assign_italian_power_helper = { power=dominant }
			}
		}
		else = {
			assign_italian_power_helper = { power=major }
		}
	}
	else_if = {
		limit = {
			check_variable = { which = italian_power value = 0.02 }
		}
		if = {
			limit = {
				check_variable = { which = italian_power value = 0.05 }
			}
			assign_italian_power_helper = { power=regional }
		}
		else = {
			assign_italian_power_helper = { power=local }
		}
	}
	else = {
		assign_italian_power_helper = { power=insignificant }
	}
	multiply_variable = { which = italian_power value = 100 } #display
}
assign_italian_power_helper = {
	if = {
		limit = {
			NOT = { has_country_flag = italian_power }
		}
		set_country_flag = italian_power
		add_country_modifier = { name = $power$_italian_power duration = -1 }
		country_event = { id = italian_power.3 }
	}
	else_if = {
		limit = {
			NOT = {
				has_country_modifier = $power$_italian_power
			}
		}
		remove_country_modifier = insignificant_italian_power
		remove_country_modifier = local_italian_power
		remove_country_modifier = regional_italian_power
		remove_country_modifier = major_italian_power
		remove_country_modifier = dominant_italian_power
		remove_country_modifier = menacing_italian_power
		add_country_modifier = { name = $power$_italian_power duration = -1 }
		country_event = { id = italian_power.2 }
	}
}

consiglio_x_power_big_add_effect = {
	change_variable = {
		which = "consiglio_x_power"
		value = 9
	}
	if = {
		limit = {
			check_variable = { which = "consiglio_x_power" value = 45 }
		}
		add_country_modifier = { name = "dominant_consiglio_dieci" duration = -1 }
		remove_country_modifier = "strong_consiglio_dieci"
		remove_country_modifier = "considerable_consiglio_dieci"
		remove_country_modifier = "weak_consiglio_dieci"
	}
	if = {
		limit = {
			check_variable = { which = "consiglio_x_power" value = 30 }
			NOT = { check_variable = { which = "consiglio_x_power" value = 45 } }
		}
		add_country_modifier = { name = "strong_consiglio_dieci" duration = -1 }
		remove_country_modifier = "dominant_consiglio_dieci"
		remove_country_modifier = "considerable_consiglio_dieci"
		remove_country_modifier = "weak_consiglio_dieci"
	}
	if = {
		limit = {
			check_variable = { which = "consiglio_x_power" value = 15 }
			NOT = { check_variable = { which = "consiglio_x_power" value = 30 } }
		}
		add_country_modifier = { name = "considerable_consiglio_dieci" duration = -1 }
		remove_country_modifier = "dominant_consiglio_dieci"
		remove_country_modifier = "strong_consiglio_dieci"
		remove_country_modifier = "weak_consiglio_dieci"
