### POP Estate Scaling effects ###

EstateProv_SetUnrestMain = {
	hidden_effect = {
		set_variable = { which = estate_$estate$_loyalty which = owner }
		set_variable = { which = EstateProv_UnrestSave which = EstateProv_Unrest }
		set_variable = { which = EstateProv_Unrest which = estate_$estate$_loyalty }
		set_variable = { which = estate_$estate$_loyalty value = 0 }
		
		if = {
			limit = { check_variable = { which = EstateProv_Unrest value = 0.6 } }
			set_province_flag = loss
			subtract_variable = { which = EstateProv_Unrest value = 0.6 }
		}
		else_if = {
			limit = { NOT = { check_variable = { which = EstateProv_Unrest value = 0.401 } } }
			set_province_flag = gain
			subtract_variable = { which = EstateProv_Unrest value = 0.4 }
			multiply_variable = { which = EstateProv_Unrest value = -1 }
		}
		else = { set_variable = { which = EstateProv_Unrest value = 0 } }
		
		# Effect of local autonomy
		export_to_variable = { which = EstateProv_LA value = local_autonomy }
		multiply_variable = { which = EstateProv_LA value = 0.008 }
		change_variable = { which = EstateProv_LA value = 0.2 }
		
		if = {
			limit = { has_province_flag = loss }
			clr_province_flag = loss
			
			multiply_variable = { which = EstateProv_Unrest value = 17.25 } # (4 * 1.725)^1.55 ~ 20
			exp_funct = {
				type1=which type2=value
				inp=EstateProv_Unrest exp=1.55 #1.7
			}
			set_variable = { which = EstateProv_Unrest which = exp_val }
			set_variable = { which = exp_val value = 0 }
			
			multiply_variable = { which = EstateProv_Unrest which = EstateProv_LA }
			
			set_variable = { which = EstateProv_Unrest_Inp which = EstateProv_Unrest }
			
			multiply_variable = { which = EstateProv_Unrest 		value = -1 }
			multiply_variable = { which = EstateProv_Unrest			value = 0.1 }
			multiply_variable = { which = EstateProv_Unrest			value = 10 }
			
			#reset_income_modifier = { varname=EstateProv_Unrest_Inp modname=EstateProv_UnrestLoss value=25.6 type=province }
			reset_income_modifier = { varname=EstateProv_Unrest_Inp modname=EstateProv_UnrestLoss value=12.8 type=province }
			reset_income_modifier = { varname=EstateProv_Unrest_Inp modname=EstateProv_UnrestLoss value=6.4 type=province }
			reset_income_modifier = { varname=EstateProv_Unrest_Inp modname=EstateProv_UnrestLoss value=3.2 type=province }
			reset_income_modifier = { varname=EstateProv_Unrest_Inp modname=EstateProv_UnrestLoss value=1.6 type=province }
			reset_income_modifier = { varname=EstateProv_Unrest_Inp modname=EstateProv_UnrestLoss value=0.8 type=province }
			reset_income_modifier = { varname=EstateProv_Unrest_Inp modname=EstateProv_UnrestLoss value=0.4 type=province }
			reset_income_modifier = { varname=EstateProv_Unrest_Inp modname=EstateProv_UnrestLoss value=0.2 type=province }
			reset_income_modifier = { varname=EstateProv_Unrest_Inp modname=EstateProv_UnrestLoss value=0.1 type=province }
		}
		else_if = {
			limit = { has_province_flag = gain }
			clr_province_flag = gain
			
			multiply_variable = { which = EstateProv_Unrest value = 16.62 } # (4 * 1.662)^1.7 ~ 25
			exp_funct = {
				type1=which type2=value
				inp=EstateProv_Unrest exp=1.7
			}
			set_variable = { which = EstateProv_Unrest which = exp_val }
			set_variable = { which = exp_val value = 0 }
			
			multiply_variable = { which = EstateProv_Unrest which = EstateProv_LA }
			
			set_variable = { which = EstateProv_Unrest_Inp which = EstateProv_Unrest }
			
			#multiply_variable = { which = EstateProv_Unrest 		value = 1 }
			multiply_variable = { which = EstateProv_Unrest			value = 0.1 }
			multiply_variable = { which = EstateProv_Unrest			value = 10 }
			
			#reset_income_modifier = { varname=EstateProv_Unrest_Inp modname=EstateProv_UnrestGain value=25.6 type=province }
			reset_income_modifier = { varname=EstateProv_Unrest_Inp modname=EstateProv_UnrestGain value=12.8 type=province }
			reset_income_modifier = { varname=EstateProv_Unrest_Inp modname=EstateProv_UnrestGain value=6.4 type=province }
			reset_income_modifier = { varname=EstateProv_Unrest_Inp modname=EstateProv_UnrestGain value=3.2 type=province }
			reset_income_modifier = { varname=EstateProv_Unrest_Inp modname=EstateProv_UnrestGain value=1.6 type=province }
			reset_income_modifier = { varname=EstateProv_Unrest_Inp modname=EstateProv_UnrestGain value=0.8 type=province }
			reset_income_modifier = { varname=EstateProv_Unrest_Inp modname=EstateProv_UnrestGain value=0.4 type=province }
			reset_income_modifier = { varname=EstateProv_Unrest_Inp modname=EstateProv_UnrestGain value=0.2 type=province }
			reset_income_modifier = { varname=EstateProv_Unrest_Inp modname=EstateProv_UnrestGain value=0.1 type=province }
		}
		
		if = {
			limit = { check_variable = { which = EstateProv_UnrestSave value = 0.001 } }
			if = {
				limit = { NOT = { check_variable = { which = EstateProv_Unrest value = 0.001 } } }
				#clear_income_modifier = { modname=EstateProv_UnrestGain value=25.6 type=province }
				clear_income_modifier = { modname=EstateProv_UnrestGain value=12.8 type=province }
				clear_income_modifier = { modname=EstateProv_UnrestGain value=6.4 type=province }
				clear_income_modifier = { modname=EstateProv_UnrestGain value=3.2 type=province }
				clear_income_modifier = { modname=EstateProv_UnrestGain value=1.6 type=province }
				clear_income_modifier = { modname=EstateProv_UnrestGain value=0.8 type=province }
				clear_income_modifier = { modname=EstateProv_UnrestGain value=0.4 type=province }
				clear_income_modifier = { modname=EstateProv_UnrestGain value=0.2 type=province }
				clear_income_modifier = { modname=EstateProv_UnrestGain value=0.1 type=province }
			}
		}
		else_if = {
			limit = {
				NOT = { check_variable = { which = EstateProv_UnrestSave value = 0 } }
				check_variable = { which = EstateProv_Unrest value = 0 }
			}
			#clear_income_modifier = { modname=EstateProv_UnrestLoss value=25.6 type=province }
			clear_income_modifier = { modname=EstateProv_UnrestLoss value=12.8 type=province }
			clear_income_modifier = { modname=EstateProv_UnrestLoss value=6.4 type=province }
			clear_income_modifier = { modname=EstateProv_UnrestLoss value=3.2 type=province }
			clear_income_modifier = { modname=EstateProv_UnrestLoss value=1.6 type=province }
			clear_income_modifier = { modname=EstateProv_UnrestLoss value=0.8 type=province }
			clear_income_modifier = { modname=EstateProv_UnrestLoss value=0.4 type=province }
			clear_income_modifier = { modname=EstateProv_UnrestLoss value=0.2 type=province }
			clear_income_modifier = { modname=EstateProv_UnrestLoss value=0.1 type=province }
		}
		
		init_as_zero = { variable=EstateProv_Unrest }
		
		set_variable = { which = EstateProv_UnrestSave value = 0 }
		set_variable = { which = EstateProv_LA value = 0 }
		set_variable = { which = EstateProv_Unrest_Inp value = 0 }
	}
}

EstateProv_SetUnrest = {
	every_owned_province = {
		limit = {
			has_province_flag = $estate$_control_province
		}
		EstateProv_SetUnrestMain = { estate=$estate$ }
	}
}
