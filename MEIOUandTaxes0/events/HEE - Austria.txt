#The Diet of Worms
country_event = {
	id = hee_HAB.1
	title = hee_HAB.1.t
	desc = hee_HAB.1.d
	picture = LIBERUM_VETO_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = HAB
		is_emperor = yes
		is_year = 1480
		NOT = {
			is_religion_enabled = protestant
		}
		NOT = {
			hre_reform_level = 3
		}
		imperial_influence = 10
	}
	
	mean_time_to_happen = { years = 10 }
	
	option = {
		name = hee_HAB.1.a
		add_imperial_influence = 20
		subtract_stability_1 = yes
		if = {
			limit = {
				61 = {
					is_part_of_hre = yes
				}
			}
			61 = {
				# add_base_tax = 1
			}
		}
		add_country_modifier = {
			name = hee_diet_worms
			duration = 7300
		}
	}
	option = {
		name = hee_HAB.1.b
		add_imperial_influence = -20
		every_country = {
			limit = {
				is_part_of_hre = yes
			}
			add_opinion = {
				who = HAB
				modifier = hee_diet_worms_opinion
			}
		}
	}
}

#Rise of the Ottomans in Europe
country_event = {
	id = hee_HAB.2
	title = hee_HAB.2.t
	desc = hee_HAB.2.d
	picture = HORDE_ON_HORSEBACK_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = HAB
		TUR = {
			num_of_cities = 12
		}
		NOT = {
			any_owned_province = {
				NOT = {
					continent = europe
				}
			}
		}
		OR = {
			any_owned_province = {
				any_neighbor_province = {
					owned_by = TUR
				}
			}
			any_subject_country = {
				any_owned_province = {
					any_neighbor_province = {
						owned_by = TUR
					}
				}
			}
		}
	}
	
	mean_time_to_happen = { months = 6 }
	
	option = {
		name = hee_HAB.2.a
		ai_chance = {
			factor = 100
		}
		add_mil_power = 50
		add_army_tradition = 5.0
		reverse_add_opinion = {
			who = TUR
			modifier = hee_defense_of_christendom
		}
	}
	option = {
		name = hee_HAB.2.b
		ai_chance = {
			factor = 0
		}
		add_dip_power = 25
		add_treasury = 10
	}
}

#The Austrian Baroque
country_event = {
	id = hee_HAB.3
	title = hee_HAB.3.t
	desc = hee_HAB.3.d
	picture = COURT_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = HAB
		is_year = 1650
		NOT = {
			is_year = 1730
		}
		government = monarchy
		num_of_cities = 10
		legitimacy = 80
		stability = 1
	}
	
	mean_time_to_happen = {
		years = 50
		modifier = {
			factor = 0.8
			stability = 2
		}
		modifier = {
			factor = 0.8
			legitimacy = 100
		}
	}
	
	option = {
		name = hee_HAB.3.a
		ai_chance = {
			factor = 85
		}
		add_country_modifier = {
			name = hee_austrian_baroque
			duration = -1
		}
		custom_tooltip = hee_HAB.3.tt
	}
	option = {
		name = hee_HAB.3.b
		ai_chance = {
			factor = 15
		}
		add_stability_1 = yes
		add_adm_power = 10
		add_dip_power = 10
	}
}

	}
}

#Build Beautiful Palaces
country_event = {
	id = hee_HAB.5
	title = hee_HAB.5.t
	desc = hee_HAB.5.d
	picture = GREAT_BUILDING_eventPicture
	
	trigger = {
		tag = HAB
		has_country_modifier = hee_austrian_baroque
		NOT = {
			has_country_modifier = hee_baroque_building_timer
		}
	}
	
	mean_time_to_happen = { years = 7 }
	
	option = {
		name = hee_HAB.5.a
		ai_chance = {
			factor = 75
			modifier = {
				factor = 25
				num_of_loans = 1
				NOT = {
					num_of_loans = 5
				}
			}
			modifier = {
				factor = 0
				num_of_loans = 5
			}
		}
		hidden_effect = {
			add_country_modifier = {
				name = hee_baroque_building_timer
				duration = 365
				hidden = yes
			}
		}
		add_treasury = -50
		add_prestige = 5
		random_owned_province = {
			limit = {
				development = 16
			}
			# add_base_tax = 1
		}
	}
	
	option = {
		name = hee_HAB.5.b
		ai_chance = {
			factor = 25
			modifier = {
				factor = 75
				num_of_loans = 1
				NOT = {

				}
			}
			modifier = {
				factor = 100

			}

	}
}
