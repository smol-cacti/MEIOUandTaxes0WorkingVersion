country_event = { #01 - The loss of Jerusalem
	
	id = dg_holylands.001
	title = dg_holylands.001.t
	desc = dg_holylands.001.d
	picture = all_refugees
	
	trigger = {
		has_country_flag = holy_land_jerusalem
		NOT = { owns_or_non_sovereign_subject_of = 379 }
		OR = {
			religion_group = jewish_group
			AND = {
				religion_group = christian
				NOT = { religion = reformed }
				NOT = { religion = protestant }
			}
			AND = {
				religion_group = muslim
				NOT = { religion = druze }
			}
		}
		piety = -0.60
	}
	
	mean_time_to_happen = { months = 3 }
	
	immediate = {
		clr_country_flag = holy_land_jerusalem
		if = { #Important to that religion
			limit = { religion_group = jewish_group }
			hidden_effect = { change_variable = { which = jerusalem value = 1 } }
		}
		if = { #Important to that nation
			limit = { 379 = { is_core = ROOT } }
			hidden_effect = { change_variable = { which = jerusalem value = 1 } }
		}
		if = { #Lost to heathens
			limit = { 379 = { owner = { NOT = { religion_group = ROOT } } } }
			hidden_effect = { change_variable = { which = jerusalem value = 1 } }
			custom_tooltip = lost_holyland_heathen
		}
		if = { #Lost to coreligionists
			limit = { 379 = { owner = { religion = ROOT } } }
			hidden_effect = { change_variable = { which = jerusalem value = -1 } }
			custom_tooltip = lost_holyland_coreligionist
		}
	}
	
	option = {
		name = dg_holylands.001.a #We have failed our task
		if = {
			limit = { check_variable = { which = jerusalem value = 2 } }
			if = {
				limit = { not_have_religious_scandal = yes }
				if = {
					limit = { piety = 0.20 }
					add_country_modifier = { name = religious_scandal_country_strong duration = 2000 }
				}
				else_if = {
					limit = { piety = -0.20 }
					add_country_modifier = { name = religious_scandal_country duration = 2000 }
				}
				else = {
					add_country_modifier = { name = religious_scandal_country_weak duration = 2000 }
				}
			}
			else = {
				if = {
					limit = { patriarch_authority = 0.80 }
					add_country_modifier = { name = unhappy_church_powerful duration = 2000 }
				}
				else_if = {
					limit = { patriarch_authority = 0.60 }
					add_country_modifier = { name = unhappy_church_strong duration = 2000 }
				}
				else_if = {
					limit = { patriarch_authority = 0.40 }
					add_country_modifier = { name = unhappy_church_moderate duration = 2000 }
				}
				else_if = {
					limit = { patriarch_authority = 0.20 }
					add_country_modifier = { name = unhappy_church_weak duration = 2000 }
				}
				else = {
					add_country_modifier = { name = unhappy_church_impotent duration = 2000 }
				}
			}
		}
		else_if = {
			limit = { is_variable_equal = { which = jerusalem value = 1 } }
			if = {
				limit = { not_have_religious_scandal = yes }
				if = {
					limit = { piety = 0.20 }
					add_country_modifier = { name = religious_scandal_country_strong duration = 1000 }
				}
				else_if = {
					limit = { piety = -0.20 }
					add_country_modifier = { name = religious_scandal_country duration = 1000 }
				}
				else = {
					add_country_modifier = { name = religious_scandal_country_weak duration = 1000 }
				}
			}
			else = {
				if = {
					limit = { patriarch_authority = 0.80 }
					add_country_modifier = { name = unhappy_church_powerful duration = 1000 }
				}
				else_if = {
					limit = { patriarch_authority = 0.60 }
					add_country_modifier = { name = unhappy_church_strong duration = 1000 }
				}
				else_if = {
					limit = { patriarch_authority = 0.40 }
					add_country_modifier = { name = unhappy_church_moderate duration = 1000 }
				}
				else_if = {
					limit = { patriarch_authority = 0.20 }
					add_country_modifier = { name = unhappy_church_weak duration = 1000 }
				}
				else = {
					add_country_modifier = { name = unhappy_church_impotent duration = 1000 }
				}
			}
		}
		else = {
			if = {
				limit = { have_religious_scandal = yes }
				if = {
					limit = { patriarch_authority = 0.80 }
					add_country_modifier = { name = unhappy_church_powerful duration = 500 }
				}
