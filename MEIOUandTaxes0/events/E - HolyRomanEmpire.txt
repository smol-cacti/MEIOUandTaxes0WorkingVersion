country_event = {
	id = holyromanempire.3
	title = "holyromanempire.3.n"
	desc = "holyromanempire.3.t"
	picture = HRE_eventPicture
	
	trigger = {
		is_at_war = no	#FB 2011.07.28
		OR = { has_regency = no has_consort_regency = yes }	#FB 2011.07.28
		prestige = 0	#FB 2011.07.28
		stability = 0	#FB 2011.07.28
		capital_scope = { is_part_of_hre = yes }
		government = monarchy
		has_country_modifier = title_5
		any_owned_province = {
			is_core = ROOT
			any_neighbor_province = {
				NOT = { owned_by = ROOT }
				NOT = { is_core = ROOT }
				culture_group = ROOT
				religion_group = ROOT	#FB 2011.07.28
				is_part_of_hre = yes
				NOT = { owned_by = emperor }
				owner = {
					NOT = { num_of_cities = ROOT }
					NOT = { prestige = ROOT }
				}
			}
		}
	}
	
	mean_time_to_happen = {
		years = 40	#FB was 120
		
		
		modifier = {
			factor = 1.5
			NOT = { DIP = 3 }
		}
		modifier = {
			factor = 1.5
			NOT = { DIP = 5 }
		}
		modifier = {
			factor = 0.5
			DIP = 6
		}
		modifier = {
			factor = 1.2
			prestige = 20	#FB 2011.07.28
		}
		modifier = {
			factor = 1.2
			prestige = 40	#FB 2011.07.28
		}
		modifier = {
			factor = 1.2
			stability = 2	#FB 2011.07.28
		}
		modifier = {
			factor = 1.2
			stability = 3	#FB 2011.07.28
		}
	}
	
	option = {
		name = "OPT.YES3"
		random_owned_province = {
			limit = {
				is_core = ROOT
				any_neighbor_province = { NOT = { owned_by = ROOT } }
			}	#FB 2011.07.28
			random_neighbor_province = {
				limit = {
					NOT = { owned_by = ROOT }
					NOT = { is_core = ROOT }
					culture_group = ROOT
					religion_group = ROOT	#FB 2011.07.28
					is_part_of_hre = yes
					NOT = { owned_by = emperor }
					owner = {
						NOT = { num_of_cities = ROOT }
						NOT = { prestige = ROOT }	#FB 2011.07.28
					}
				}
				add_core = ROOT
			}
		}
	}
}
