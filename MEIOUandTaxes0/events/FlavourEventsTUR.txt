province_event = { #Loss of $PROVINCENAME$
	
	id = flavor_tur.1007
	title = flavor_tur.1007.t
	desc = flavor_tur.1007.d
	picture = muslim_army
	
	trigger = {
		controlled_by = turkish_ghazi
		had_province_flag = { flag = turkish_ghazi days = 180 }
		has_siege = no
		NOT = { owner = { war_with = TUR } }
		NOT = { owner = { war_with = OTT } }
		NOT = { owned_by = TUR }
		NOT = { owned_by = OTT }
	}
	
	mean_time_to_happen = { months = 3 }
	
	option = {
		name = flavor_tur.1007.a #Cede the province
		ai_chance = { factor = 100 }
		owner = {
			disband_rebels = turkish_ghazi
			every_owned_province = {
				clr_province_flag = turkish_ghazi
				clr_province_flag = turkish_adventurers
			}
		}
		if = {
			limit = {
				exists = TUR
				NOT = { owner = { tag = TUR } }
			}
			cede_province = TUR
			hidden_effect = { TUR = { add_manpower = 1 add_treasury = 10 } }
		}
		if = {
			limit = {
				NOT = { exists = TUR }
				exists = OTT
				NOT = { owner = { tag = OTT } }
			}
			cede_province = OTT
			hidden_effect = { OTT = { add_manpower = 1 add_treasury = 10 } }
		}
	}
	option = {
		name = flavor_tur.1007.b #Fight the Turks for it
		ai_chance = { factor = 0 }
		if = {
			limit = {
				OR = {
					owner = { truce_with = TUR }
					owner = { truce_with = OTT }
				}
			}
			owner = { subtract_stability_2 = yes }
		}
		if = {
			limit = {
				exists = TUR
				NOT = { owner = { tag = TUR } }
			}
			owner = { declare_war = TUR }
		}
		if = {
			limit = {
				NOT = { exists = TUR }
				exists = OTT
				NOT = { owner = { tag = OTT } }
			}
			owner = { declare_war = OTT }
		}
	}
}
