# The Marathi Revolt
country_event = {
	id = flavor_dlh.1
	title = "flavor_dlh.EVTNAME1"
	desc = "flavor_dlh.EVTDESC1"
	picture = ANGRY_MOB_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		is_year = 1625
		NOT = { exists = MAR }
		OR = {
			NOT = { religion = hinduism }
			NOT = { accepted_culture = marathi }
		}
		NOT = { stability = 0 }
		num_of_cities = 5
		any_owned_province = {
			OR = {
				province_id = 530 # Maharashtra
				province_id = 1587 # Bijapur
				province_id = 2243 # Ahmadnagar
				province_id = 2569 # Sahyadri
				province_id = 2670 # Kamphaeng Phet
				province_id = 2679 # Kolhapur
				culture = marathi
			}
			religion = hinduism
		}
	}
	
	mean_time_to_happen = { years = 16 }
	
	option = {
		name = "flavor_dlh.EVTOPTA1"
		every_owned_province = {
			limit = {
				OR = {
					province_id = 530 # Maharashtra
					province_id = 1587 # Bijapur
					province_id = 2243 # Ahmadnagar
					province_id = 2569 # Sahyadri
					province_id = 2670 # Kamphaeng Phet
					province_id = 2679 # Kolhapur
					culture = marathi
				}
			}
			add_core = MAR
			hidden_effect = {
				set_variable = { which = added_unrest value = 10 }
				add_base_unrest = yes
			}
			custom_tooltip = added_unrest_10
		}
	}
	option = {
		name = "flavor_dlh.EVTOPTB1"
		every_owned_province = {
			limit = {
				OR = {
					province_id = 530 # Maharashtra
					province_id = 1587 # Bijapur
					province_id = 2243 # Ahmadnagar
					province_id = 2569 # Sahyadri
					province_id = 2670 # Kamphaeng Phet
					province_id = 2679 # Kolhapur
					culture = marathi
				}
			}
			add_local_autonomy = 100
		}
	}
}

# The Lodi Dynasty
country_event = {
	id = flavor_dlh.3
	title = "flavor_dlh.EVTNAME3"
	desc = "flavor_dlh.EVTDESC3"
	picture = COURT_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = DLH
		is_year = 1452
		NOT = { is_year = 1553 }
		dynasty = "Lodi"
		OR = {
			NOT = { dip = 0 }
			NOT = { mil = 0 }
			NOT = { adm = 0 }
		}
	}
	
	mean_time_to_happen = { years = 16 }
	
	option = {
		name = "flavor_dlh.EVTOPTA3"
		random_list = {
			30 = {}
			40 = {
				subtract_stability_points_50 = yes
			}
			30 = {
				subtract_stability_1 = yes
			}
		}
		every_owned_province = {
			limit = { is_core = MAR }
			hidden_effect = {
				set_variable = { which = added_unrest value = 2 }
				add_base_unrest = yes
			}
			custom_tooltip = added_unrest_2
		}
		add_inflation = 1
		set_country_flag = dlh_lodi_flag
	}
}

# The Marathi Revolt
country_event = {
	id = flavor_dlh.5
	title = "flavor_dlh.EVTNAME1"
	desc = "flavor_dlh.EVTDESC1"
	picture = ANGRY_MOB_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		is_year = 1625
		NOT = { exists = MAR }
		OR = {
			NOT = { religion = hinduism }
			NOT = { accepted_culture = marathi }
		}
		NOT = { stability = 0 }
		num_of_cities = 5
		any_owned_province = {
			OR = {
				province_id = 530 # Maharashtra
				province_id = 1587 # Bijapur
				province_id = 2243 # Ahmadnagar
				province_id = 2569 # Sahyadri
				province_id = 2670 # Kamphaeng Phet
				province_id = 2679 # Kolhapur
				culture = marathi
			}
			religion = hinduism
		}
	}
	
	mean_time_to_happen = {
		years = 166
		modifier = {
			factor = 0.75
			NOT = { accepted_culture = marathi }
		}
		modifier = {
			factor = 0.75
			NOT = { primary_culture = marathi }
		}
		modifier = {
			factor = 0.75
			num_of_owned_provinces_with = {
				value = 2
				culture = marathi
				religion = hinduism
			}
		}
		modifier = {
			factor = 0.75
			num_of_owned_provinces_with = {
				value = 4
				culture = marathi
				religion = hinduism
			}
		}
		modifier = {
			factor = 0.75
			num_of_owned_provinces_with = {
				value = 6
				culture = marathi
				religion = hinduism
			}
		}
		modifier = {
			factor = 0.75
			num_of_owned_provinces_with = {
				value = 8
				culture = marathi
				religion = hinduism
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			every_owned_province = {
				limit = {
					religion = hinduism
					culture = marathi
				}
				discover_country = MAR
				add_core = MAR
			}
			release = MAR
			regiongroup = {
				region = {
					limit = {
						is_city = yes
						culture = marathi
					}
					add_claim = MAR
				}
			}
		}
	}
	
	option = {
		name = "flavor_dlh.EVTOPTA1"
		MAR = {
			declare_war_with_cb = {
				who = ROOT
				casus_belli = cb_independence_war
			}
		}
		hidden_effect = {
			MAR = {
				country_event = { id = flavor_mar.1 days = 1 }
			}
		}
	}
}
