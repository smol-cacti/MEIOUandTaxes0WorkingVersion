# The Ambrosian Republic
country_event = {
	id = flavor_mlo.1
	title = flavor_mlo.1.t
	desc = flavor_mlo.1.d
	picture = DEBATE_REPUBLICAN_eventPicture
	
	fire_only_once = yes
	is_triggered_only = yes
	
	trigger = {
		tag = MLO
		NOT = { has_country_flag = ambrosian_republic }
		government = monarchy
		OR = {
			AND = {
				has_new_dynasty = yes
				has_regency = no
				has_consort_regency = no
			}
			any_known_country = {
				succession_claim = ROOT
			}
		}
		is_lesser_in_union = no
		NOT = { is_year = 1550 }
	}
	
	option = {
		name = flavor_mlo.1.a
		ai_chance = { factor = 90 }
		oligarchisation_effect = yes
		add_republican_tradition = 5
		every_country = {
			limit = {
				religion = catholic
				num_of_cities = 5
				NOT = {
					tag = MLO
				}
				OR = {
					government = monarchy
					tag = VEN
					tag = PAP
				}
				OR = {
					culture_group = low_germanic
					culture_group = high_germanic
					culture_group = latin
					culture_group = langue_d_oil
					culture_group = langue_d_oc
					is_neighbor_of = ROOT
				}
			}
			
			add_opinion = {
				who = ROOT
				modifier = mlo_against_ambrosian
			}
			if = {
				limit = {
					government = monarchy
				}
				
				add_casus_belli = {
					target = MLO
					type = cb_change_government
					months = 120
				}
			}
		}
		set_country_flag = ambrosian_republic
	}
	option = {
		name = flavor_mlo.1.b
		ai_chance = { factor = 10 }
		subtract_stability_2 = yes
		add_legitimacy = 25
	}
}
