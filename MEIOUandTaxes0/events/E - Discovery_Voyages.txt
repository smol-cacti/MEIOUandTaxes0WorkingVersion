# discovery_voyages.0101 : Gonçalo Velho Cabral's voyage
country_event = {
	id = discovery_voyages.0101
	title = "discovery_voyages.0101.name"
	desc = "discovery_voyages.0101.desc"
	picture = CARAVELS_eventPicture
	
	trigger = {
		tag = POR
		NOT = { has_global_flag = goncalo_cabral }
		NOT = { has_discovered = 367 }	# Azores
		dip_tech = 15
		treasury = 50
		367 = { is_empty = yes }
	}
	
	mean_time_to_happen = {
		years = 83
		
		modifier = {
			ai = no
			factor = 1.1
		}
	}
	
	immediate = {
		set_global_flag = goncalo_cabral	# Flag prevents event firing again
	}
	
	option = {
		name = "discovery_voyages.0001.optd"
		ai_chance = { factor = 100 }
		add_treasury = -50
		discover_province = 1668	# Iberian Shelf
		discover_province = 1669	# Sea of Azores
		discover_province = 367		# Os Acores
		367 = { # Os Acores
			if = {
				limit = {
					non_colonial_owner_trigger = yes
				}
				cede_province = ROOT
				create_colony = 100
				change_culture = ROOT
				change_religion = ROOT
			}
			else = {
				add_permanent_claim = ROOT
			}
		}
	}
	
	option = {
		name = "discovery_voyages.0001.optc"
		ai_chance = { factor = 0 }
		discover_province = 1668	# Iberian Shelf
		discover_province = 1669	# Sea of Azores
		discover_province = 367		# Os Acores
	}
}
