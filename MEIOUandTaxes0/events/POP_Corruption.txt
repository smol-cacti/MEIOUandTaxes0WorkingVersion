country_event = { ### Bribery
	id = POP_Corruption.001
	title = "POP_Corruption.001.t"
	desc = "POP_Corruption.001.d"
	is_triggered_only = yes
	hidden = no
	picture = CORRUPTION_eventPicture
	
	immediate = {
		hidden_effect = {
			set_variable = { which = ai_corruption_choice 	value = 0 } ### Used to inform AI as to whether to take the corruption or not
			random_list = { ### Monarch Point Volume
				10 = { set_variable = { which = admin_cost_a 	value = 20 } subtract_variable = { which = ai_corruption_choice		value = 4 } }
				20 = { set_variable = { which = admin_cost_a 	value = 40 } subtract_variable = { which = ai_corruption_choice		value = 2 } }
				40 = { set_variable = { which = admin_cost_a 	value = 100 } }
				20 = { set_variable = { which = admin_cost_a 	value = 150 } change_variable = { which = ai_corruption_choice		value = 3 } }
				10 = { set_variable = { which = admin_cost_a 	value = 200 } change_variable = { which = ai_corruption_choice		value = 5 } }
			}
			random_list = { ### Stability Hit
				10 = { set_variable = { which = stability_hit 	value = 10 } subtract_variable = { which = ai_corruption_choice		value = 6 } }
				25 = { set_variable = { which = stability_hit 	value = 20 } subtract_variable = { which = ai_corruption_choice		value = 2 } }
				30 = { set_variable = { which = stability_hit 	value = 40 } change_variable = { which = ai_corruption_choice 		value = 3 } }

				10 = { set_variable = { which = stability_hit 	value = 70 } change_variable = { which = ai_corruption_choice 		value = 6 } }
			}
			random_list = { ### Stability Modifier Malus
				10 = { set_variable = { which = fight_corruption_malus 	value = 10 } subtract_variable = { which = ai_corruption_choice		value = 4 } }
				20 = { set_variable = { which = fight_corruption_malus 	value = 15 } subtract_variable = { which = ai_corruption_choice		value = 2 } }
				40 = { set_variable = { which = fight_corruption_malus 	value = 25 } change_variable = { which = ai_corruption_choice 		value = 1 } }
				20 = { set_variable = { which = fight_corruption_malus 	value = 35 } change_variable = { which = ai_corruption_choice 		value = 3 } }
				10 = { set_variable = { which = fight_corruption_malus 	value = 50 } change_variable = { which = ai_corruption_choice 		value = 5 } }
			}
			if = {
				limit = {
					ai = no
				}
				#set_variable = { 	  which = BU_corruption_current 		value = 0 }
				set_variable = {   which = BU_corruption_current 		which = privilege_corruption_BU }
				multiply_variable = {   which = BU_corruption_current			value = 0.01 }
				
				#set_variable = {      which = BU_corruption_cost_1 			value = 0 }
				set_variable = {   which = BU_corruption_cost_1 			value = 5 }
				#	multiply_variable = { which = BU_corruption_cost_1 			value = 2 }
				multiply_variable = {   which = BU_corruption_cost_1			value = 0.01 }
			}
		}
	}
	
	option = { ### Accept it
		name = "POP_Corruption.001.a"
		ai_chance = {
			factor = 1
			modifier = {
				factor = 1000
				check_variable = { which = ai_corruption_choice 	value = 0 }
			}
		}
		add_country_modifier = {
			name = accept_bribe_corruption
			duration = 3650
		}
		BU_corruption_added_1_tooltip_effect = yes
		hidden_effect = {
			add_country_modifier = {
				name = recently_corrupted
				duration = 1825
				hidden = yes
			}
		}
		
		if = {
			limit = {
				has_country_modifier = BU_bribery_common_3
				NOT = { has_country_modifier = BU_bribery_common_4 }
			}
			remove_country_modifier = BU_bribery_common_3
			add_country_modifier = { name = BU_bribery_common_4 duration = -1 }
			remove_country_modifier = BU_bribery_common_4
			add_country_modifier = { name = BU_bribery_common_4 duration = -1 hidden = yes }
		}
		if = {
			limit = {
				has_country_modifier = BU_bribery_common_2
				NOT = { has_country_modifier = BU_bribery_common_3 }
			}
			remove_country_modifier = BU_bribery_common_2
			add_country_modifier = { name = BU_bribery_common_3 duration = -1 }
			remove_country_modifier = BU_bribery_common_3
			add_country_modifier = { name = BU_bribery_common_3 duration = -1 hidden = yes }
		}
		if = {
			limit = {
				has_country_modifier = BU_bribery_common_1
				NOT = { has_country_modifier = BU_bribery_common_2 }
			}
			remove_country_modifier = BU_bribery_common_1
			add_country_modifier = { name = BU_bribery_common_2 duration = -1 }
			remove_country_modifier = BU_bribery_common_2
			add_country_modifier = { name = BU_bribery_common_2 duration = -1 hidden = yes }
		}
		if = {
			limit = {
				NOT = { has_country_modifier = BU_bribery_common_1 }
				NOT = { has_country_modifier = BU_bribery_common_2 }
				NOT = { has_country_modifier = BU_bribery_common_3 }
				NOT = { has_country_modifier = BU_bribery_common_4 }
			}
			add_country_modifier = { name = BU_bribery_common_1 duration = -1 }
			remove_country_modifier = BU_bribery_common_1
			add_country_modifier = { name = BU_bribery_common_1 duration = -1 hidden = yes }
		}
		hidden_effect = {
			#if = { ### Variables set to 0 after calculations are done
				#limit = {
				#	NOT = { has_global_flag = show_for_developers }
				#}
				set_variable = { which = admin_cost_a				value = 0 }
				set_variable = { which = stability_hit				value = 0 }
				set_variable = { which = ai_corruption_choice		value = 0 }
				set_variable = { which = fight_corruption_malus		value = 0 }
				set_variable = { which = BU_corruption_current		value = 0 }
				set_variable = { which = BU_corruption_cost_1		value = 0 }
				set_variable = { which = admin_cost					value = 0 }
			#}
		}
	}
	
	option = { ### Fight it
		name = "POP_Corruption.001.b"
		ai_chance = { factor = 10 }
		trigger = { stability = 0 }
		
		custom_tooltip = admin_cost_a
		hidden_effect = {
			#set_variable = { which = admin_cost 	value = 0 }
			set_variable = { which = admin_cost 	which = admin_cost_a }
			scaled_admin_cost_country = yes
		}
		fight_corruption_stability = yes
		corruption_stab_modifier = yes
		hidden_effect = {
			#if = { ### Variables set to 0 after calculations are done
