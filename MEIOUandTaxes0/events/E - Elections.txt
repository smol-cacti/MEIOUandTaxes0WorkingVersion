	id = elections.700
	title = "EVTNAME700"
	desc = {
		trigger = { NOT = { has_ruler_flag = leader_has_been_pushed_out } }
		desc = "EVTDESC700"
	}

		trigger = { has_ruler_flag = leader_has_been_pushed_out }
		desc = "elections.700.d"
	}
	picture = ELECTION_REPUBLICAN_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		is_colonial_nation = no
		NOT = { has_country_flag = medici_signoria }
	}
	
	option = {
		name = "EVTOPTA700"	# Current ruler stays
		trigger = {
			has_regency = no
			NOT = { has_ruler_flag = leader_has_been_pushed_out }
		}
		ai_chance = {
			factor = 20
			modifier = {
				factor = 0
				NOT = { republican_tradition = 25 }
			}
			modifier = {
				factor = 0.5
				NOT = { republican_tradition = 50 }
			}
			modifier = {
				factor = 0.5
				NOT = { republican_tradition = 75 }
			}
			modifier = {
				factor = 2.0
				republican_tradition = 90
			}
		}
		custom_tooltip = remains_ruler
		add_scaled_republican_tradition = -10
	}
	option = {
		name = "EVTOPTB700"	# option A
		ai_chance = { factor = 20 }
		define_ruler = {
		
		}
	}
	option = {
		name = "EVTOPTC700"	# option B
		ai_chance = { factor = 20 }
		define_ruler = {
		
		}
	}
}

# Election!
country_event = {
	id = elections.706
	title = "elections.EVTNAME706"
	desc = "elections.EVTDESC706"
	picture = ELECTION_REPUBLICAN_eventPicture
	
	is_triggered_only = yes
	hidden = yes
	
	trigger = {
		is_colonial_nation = yes
	}
	
	option = {
		name = "elections.EVTOPTB706"
		if = {
			limit = {
				colonial_parent = {
					OR = {
						ai = yes
						has_country_flag = auto_colonial_elections
					}
				}
			}
			define_ruler = {
			}
		}
		else = {
			colonial_parent = {
				country_event = { id = elections.707 }
			}
		}
	}
}

	id = elections.707
	title = "elections.EVTNAME707"
	desc = "elections.EVTDESC707"
	picture = ELECTION_REPUBLICAN_eventPicture
	
	is_triggered_only = yes
	
	option = {
		name = "elections.EVTOPTA707"	# Same Candidate
		ai_chance = { factor = 10 }
		trigger = {
			has_country_flag = feudal_colonisation
		}
		FROM = {
			add_scaled_republican_tradition = -10
			add_liberty_desire = 5
		}
	}
	option = {
		name = "elections.EVTOPTB707"	# Bureaucrat Candidate
		ai_chance = { factor = 20 }
		FROM = {
			define_ruler = {
				max_random_dip = 3

			}

		}
