# Przywilej mielnicki
country_event = {
	id = hee_POL.12
	title = hee_POL.12.t
	desc = hee_POL.12.d
	picture = COURT_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		OR = {
			AND = {
				tag = POL
				NOT = { exists = PLC }
			}
			tag = PLC
		}
		is_year = 1495
		NOT = { is_year = 1530 }
	}

# Turko-Polish Tension
country_event = {
	id = flavor_pol.3488
	title = "flavor_pol.EVTNAME3488"
	desc = "flavor_pol.EVTDESC3488"
	picture = DIPLOMACY_eventPicture
	
	trigger = {
		OR = {
			tag = PLC
			AND = {
				tag = POL
				NOT = { exists = PLC }
			}
		}
		OR = {
			AND = {
				is_neighbor_of = TUR
				TUR = {
					any_owned_province = {
						any_neighbor_province = {
							owned_by = ROOT
						}
						NOT = { is_core = ROOT }
						NOT = { is_claim = ROOT }
					}
				}
			}
			any_known_country = {
				vassal_of = ROOT
				is_neighbor_of = TUR
				TUR = {
					any_owned_province = {
						any_neighbor_province = {
							owner = {
								vassal_of = ROOT
							}
						}
						NOT = { is_core = ROOT }
						NOT = { is_claim = ROOT }
					}
				}
			}
			any_known_country = {
				OR = {
					junior_union_with = ROOT
					senior_union_with = ROOT
				}
				is_neighbor_of = TUR
				TUR = {
					any_owned_province = {
						any_neighbor_province = {
							owner = {
								OR = {
									junior_union_with = ROOT
									senior_union_with = ROOT
								}
							}
						}
						NOT = { is_core = ROOT }
						NOT = { is_claim = ROOT }
					}
				}
			}
		}
		NOT = { has_country_modifier = pol_tur_tension_timer }
	}
	
	mean_time_to_happen = { years = 16 }
	
	option = {
		name = "flavor_pol.EVTOPTA3488" # We will defend Christendom
		add_country_modifier = {
			name = pol_tur_tension_timer
			duration = 365
			hidden = yes
		}
		add_yearly_manpower = 1
		if = {
			limit = {
				is_neighbor_of = TUR
			}
			TUR = {
				random_owned_province = {
					limit = {
						any_neighbor_province = {
							owned_by = ROOT
						}
						NOT = { is_core = ROOT }
						NOT = { is_claim = ROOT }
					}
					add_claim = ROOT
				}
			}
		}
		if = {
			limit = {
				NOT = { is_neighbor_of = TUR }
				any_known_country = {
					vassal_of = ROOT
					is_neighbor_of = TUR
				}
			}
			TUR = {
				random_owned_province = {
					limit = {
						any_neighbor_province = {
							owner = {
								vassal_of = ROOT
							}
						}
						NOT = { is_core = ROOT }
						NOT = { is_claim = ROOT }
					}
					add_claim = ROOT
				}
			}
		}
		if = {
			limit = {
				NOT = { is_neighbor_of = TUR }
				NOT = {
					any_known_country = {
						vassal_of = ROOT
						is_neighbor_of = TUR
					}
				}
				any_known_country = {
					OR = {
						junior_union_with = ROOT
						senior_union_with = ROOT
					}
					is_neighbor_of = TUR
				}
			}
			TUR = {
				random_owned_province = {
					limit = {
						any_neighbor_province = {
							owner = {
								OR = {
									junior_union_with = ROOT
									senior_union_with = ROOT
								}
							}
						}
						NOT = { is_core = ROOT }
						NOT = { is_claim = ROOT }
					}
					add_claim = ROOT
				}
			}
		}
	}
}
