# Queen Elizabeth I (1533-1603)
country_event = {
	id = women_history_events.1
	title = women_history_events.1.t
	desc = women_history_events.1.d
	picture = wihgfx_FEMALE_RULER_eventPicture
	fire_only_once = yes
	
	
	trigger = {
		is_year = 1533
		NOT = {
			is_year = 1603
		}
		OR = {
			tag = ENG
			tag = GBR
		}
		is_female = no
		has_regency = no
		has_heir = no
		is_lesser_in_union = no
		government = monarchy
		religion = catholic
		any_owned_province = {
			religion = protestant
			is_in_capital_area = yes
		}
	}
	
	mean_time_to_happen = {
		months = 200
	}
	
	
	option = {
		name = women_history_events.1.a
		change_religion = protestant
		define_ruler = {
			name = "Elizabeth"
			dynasty = ROOT
			adm = 6
			dip = 6
			mil = 5
			female = yes
		}
		if = {
			limit = {
				has_dlc = "Rights of Man"
			}
			hidden_effect = { add_ruler_personality = inspiring_leader_personality }
		}
	}
	option = {
		name = women_history_events.1.b
		add_adm_power = 100
		add_dip_power = 100
		add_mil_power = 100
		#add_meritocracy_effect = yes
		define_advisor = {
			culture = english
			female = yes
			type = statesman #spymaster
			skill = 3
			discount = yes
			name = "Elizabeth Tudor"
		}
	}
}
