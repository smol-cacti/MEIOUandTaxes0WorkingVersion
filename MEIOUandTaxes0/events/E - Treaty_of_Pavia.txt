

# von Wittelsbach - Treaty of Pavia (Wittelsbacher Hausvertrag von Pavia)
# If the male line in one of the branches of the von Wittelsbach dies out (at least for the Bavarian and Palatinatean branches), their lands get inherited by the other members of the respective (sub)branch
# so if the line in Bayern-Straubing ends, the Province gets inherited by either Bayern-München or Bayern-Landshut (if they exist)
# If all Bavarian branches die off, their lands get inherited by the Palatinate, and vice-versa
# If they have an heir of a different dynasty or a female heir they can protest and a war of succession will ensue.
# see http://en.wikipedia.org/wiki/War_of_the_Succession_of_Landshut for reference

country_event = {
	id = treaty_pavia.1000
	title = "treaty_pavia.1000.t"
	desc = "EVT_AI"
	picture = BAVARIA_eventPicture
	
	is_triggered_only = yes
	
	hidden = yes
	
	trigger = {
		NOT = { has_country_flag = treaty_of_pavia_ignored }
		government = monarchy
		OR = {
			tag = BAV		# Bavaria
			tag = BAW		# Ingolstadt
			tag = BAX		# Landshut
			tag = BAY		# Munchen
			tag = BAZ		# Straubing
			tag = PAL		# Palatinate
		}
		is_at_war = no
		OR = {
			#has_ruler = "Interregnum"
			AND = {
				is_female = yes
				has_consort_regency = no
			}
			NOT = { dynasty = "von Wittelsbach" }
			has_new_dynasty = yes
			has_country_flag = bav_no_heir
			AND = {
				has_regency = yes
				OR = {
					has_heir = no
					has_foreign_heir = yes
				}
			}
		}
		any_known_country = {
			#exists = yes
			government = monarchy
			NOT = { tag = ROOT }
			dynasty = "von Wittelsbach"
		}
	}
	
	immediate = {
		hidden_effect = {
			if = {
				limit = {
					is_subject_other_than_tributary_trigger = yes
					overlord = {
						#exists = yes
						dynasty = "von Wittelsbach"
					}
				}
				every_core_province = {
					overlord = {
						if = {
							limit = {
								dynasty = "von Wittelsbach"
							}
							PREV = { add_core = PREV }
						}
					}
				}
				if = {
					limit = {
						ai = yes		# Inheriting only for AI, don't want to kick human players out of the game
					}
					overlord = {
						if = {
							limit = {
								dynasty = "von Wittelsbach"
							}
							inherit = ROOT
						}
					}
				}
			}
			if = {
				limit = {
					NOT = { dynasty = "von Wittelsbach" }
					NOT = {
						is_subject_other_than_tributary_trigger = yes
						overlord = {
							#exists = yes
							dynasty = "von Wittelsbach"
						}
					}
				}
				ROOT = {
					country_event = { id = treaty_pavia.1010 }
				}
			}
			if = {
				limit = {
					dynasty = "von Wittelsbach"
					NOT = {
						is_subject_other_than_tributary_trigger = yes
						overlord = {
							#exists = yes
							dynasty = "von Wittelsbach"
						}
					}
					# since it's still under Wittelsbach rule, just not the right branch, only other bavarian Wittelsbach rulers have claims, other Wittelsbachs are denied a claim on PU
					any_known_country = {
						#exists = yes
						dynasty = "von Wittelsbach"
						NOT = { tag = ROOT }
						OR = {
							tag = BAV		# Bavaria
							tag = BAW		# Ingolstadt
							tag = BAX		# Landshut
							tag = BAY		# Munchen
							tag = BAZ		# Straubing
							tag = PAL		# Palatinate
						}
					}
				}
				ROOT = {
					country_event = { id = treaty_pavia.1020 }
				}
			}
		}
	}
	
	
	option = {
		name = "EVT_AI"
	}
}

province_event = {
	id = treaty_pavia.3001
	title = "treaty_pavia.3001.t"
	desc = "EVT_AI"
	picture = BAVARIA_eventPicture
	
	is_triggered_only = yes
	
	hidden = yes
	
	
	immediate = {
		hidden_effect = {
			# is Palatinate and Bavaria is eligible, Bavaria gets it all
			if = {
				limit = {
					FROM = { tag = PAL }
					any_known_country = {
						tag = BAV
						#exists = yes
						government = monarchy
						dynasty = "von Wittelsbach"
					}
				}
				if = {
					limit = {
						owned_by = FROM
					}
					cede_province = BAV
					add_claim = BAV
				}
				if = {
					limit = {
						is_core = FROM
					}
					add_core = BAV
					remove_core = FROM
				}
			}
			# is Palatinate and Bavaria is not eligible (which means, any of the "sub"-bavarias must exist, and thus Bavaria can't exist) the land is split up among whichever Bavarian minors exist
			if = {
				limit = {
					FROM = { tag = PAL }
					NOT = {
						any_known_country = {
							tag = BAV
							#exists = yes
							government = monarchy
							dynasty = "von Wittelsbach"
						}
					}
				}
				if = {
					limit = {
						OR = {
							is_core = FROM
							owned_by = FROM
						}
					}
					random_known_country = {
						limit = {
							#exists = yes
							OR = {
								tag = BAW
								tag = BAX
								tag = BAY
								tag = BAZ
							}
							government = monarchy
							dynasty = "von Wittelsbach"
						}
						if = {
							limit = {
								ROOT = { owned_by = FROM }
							}
							ROOT = { cede_province = PREV }
							ROOT = { add_claim = PREV }
						}
						if = {
							limit = {
								ROOT = { is_core = FROM }
							}
							ROOT = { add_core = PREV }
							ROOT = { remove_core = FROM }
						}
					}
				}
			}
			# if is BAV, PAL gets it all
			if = {
				limit = {
					FROM = { tag = BAV }
					any_known_country = {
						#exists = yes
						tag = PAL
						government = monarchy
						dynasty = "von Wittelsbach"
					}
				}
				if = {
					limit = {
						is_core = FROM
					}
					add_core = PAL
					remove_core = FROM
				}
				if = {
					limit = {
						owned_by = FROM
					}
					cede_province = PAL
					add_claim = PAL
				}
			}
			# if BAW, BAX, BAY, BAZ and any other of those three is eligible, they'll get a chunk
			if = {
				limit = {
					FROM = {
						OR = {
							tag = BAW
							tag = BAX
							tag = BAY
							tag = BAZ
						}
					}
					any_known_country = {
						#exists = yes
						NOT = { tag = FROM }
						OR = {
							tag = BAW
							tag = BAX
							tag = BAY
							tag = BAZ
						}
						government = monarchy
						dynasty = "von Wittelsbach"
					}
				}
				if = {
					limit = {
						OR = {
							is_core = FROM
							owned_by = FROM
						}
					}
					random_known_country = {
						limit = {
							#exists = yes
							OR = {
								tag = BAW
								tag = BAX
								tag = BAY
								tag = BAZ
							}
							government = monarchy
							dynasty = "von Wittelsbach"
						}
						if = {
							limit = {
								ROOT = { owned_by = FROM }
							}
							ROOT = { cede_province = PREV }
							ROOT = { add_claim = PREV }
						}
						if = {
							limit = {
								ROOT = { is_core = FROM }
							}
							ROOT = { add_core = PREV }
							ROOT = { remove_core = FROM }
						}
					}
				}
			}
			# if BAW, BAX, BAY, BAZ and only PAL eligible, PAL gets it all
			if = {
				limit = {
					FROM = {
						OR = {
							tag = BAW
							tag = BAX
							tag = BAY
							tag = BAZ
						}
					}
					NOT = {
						any_known_country = {
							#exists = yes
							NOT = { tag = FROM }
							OR = {
								tag = BAW
								tag = BAX
								tag = BAY
								tag = BAZ
							}
							government = monarchy
							dynasty = "von Wittelsbach"
						}
					}
				}
				if = {
					limit = {
						owned_by = FROM
					}
					add_claim = PAL
					cede_province = PAL
				}
				if = {
					limit = {
						is_core = FROM
					}
					add_core = PAL
					remove_core = FROM
				}
			}
		}
	}
	
	option = {
		name = "EVT_AI"
	}
}
