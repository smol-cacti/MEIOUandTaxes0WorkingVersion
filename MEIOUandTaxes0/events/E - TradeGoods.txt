#colonial_goods.1 Wheat for European colonies
province_event = {
	id = colonial_goods.1
	title = "NEWS_FROM_COL"
	desc = "EVTDESCcolonial_goods.1"
	picture = TRADEGOODS_eventPicture
	
	trigger = {
		trade_goods = unknown
		OR = { colonysize = 300 is_city = yes }
		OR = {
			AND = {
				continent = europe
				NOT = { superregion = scandinavian_superregion }
			}
			AND = {
				continent = oceania
				OR = {
					owner = {
						is_colonial_nation = no
						capital_scope = { continent = europe }
					}
					owner = {
						is_colonial_nation = yes
						colonial_parent = { capital_scope = { continent = europe } }
					}
				}
			}
		}
	}
	
	mean_time_to_happen = { years = 5 }
	
	option = {
		name = "OPT.VERYWELL"
		change_trade_goods = wheat
		hidden_effect = { set_base_price_of_goods = yes }
	}
}
