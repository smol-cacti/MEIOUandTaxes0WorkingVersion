#Jörg Jenatsch

	id = hee_SWI.5

	desc = hee_SWI.5.d
	picture = ADVISOR_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = SWI
		is_year = 1600

		OR = {
			any_owned_province = {
				religion = protestant
			}

				religion = reformed
			}
		}
	}
	
	mean_time_to_happen = {
		years = 25

			factor = 0.5

		}
	}
	
	option = {
		name = hee_SWI.5.a
		add_army_tradition = 5.0
		define_advisor = {
			type = army_reformer
			name = "Jörg Jenatsch"
			skill = 2
			discount = yes
		}
	}
	option = {
		name = hee_SWI.5.b
		add_army_tradition = 5.0
		add_mil_power = 25
	}
}

#Bündner Wirren
country_event = {
	id = hee_SWI.6
	title = hee_SWI.6.t
	desc = hee_SWI.6.d
	picture = COUNTRY_COLLAPSE_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		
		tag = SWI
		is_year = 1550
		NOT = {
			is_year = 1680
		}
		owns_core_province = 166 # Grischun
		exists = FRA
		exists = HAB
		is_religion_enabled = protestant
	}
	
	mean_time_to_happen = {
		years = 75
		modifier = {
			factor = 0.8
			is_neighbor_of = HAB
		}
		modifier = {
			factor = 0.8
			alliance_with = FRA
		}
		modifier = {
			factor = 0.8
			is_neighbor_of = VEN
		}
		modifier = {
			factor = 0.2
			is_in_league_war = yes
		}
	}
	
	option = {
		name = hee_SWI.6.a
		166 = {
			hidden_effect = {
				set_variable = { which = added_unrest value = 5 }
				add_base_unrest = yes
			}
			custom_tooltip = added_unrest_5

			change_variable = { which = province_religion value = -14 }

		}

			target = HAB
			type = cb_insult

	fire_only_once = yes
	
	trigger = {
		tag = SWI
		is_year = 1620
		NOT = { is_year = 1680 }
	}
	
	mean_time_to_happen = {
		years = 33
		modifier = {
			factor = 0.75
			NOT = {
				emperor = {
					reverse_has_opinion = {
						who = ROOT
						value = 0
					}
				}
			}
		}
		modifier = {
			factor = 0.75
			emperor = { is_in_league_war = yes }
		}
	}
	
	option = {
		name = hee_SWI.7.a
		define_advisor = {
			type = statesman
			name = "Johann Rudolf Wettstein"
			skill = 3
			discount = yes
		}
		add_dip_power = 50
	}
	option = {
		name = hee_SWI.7.b
		add_country_modifier = {
			name = hee_wettstein
			duration = 7200
		}
		add_dip_power = 50
	}
	option = {
		name = hee_SWI.7.c
		add_dip_power = 100
	}
}
