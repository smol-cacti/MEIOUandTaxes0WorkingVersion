country_event = {
	
	id = chinese_factions.1
	title = "chinese_factions.1.n"				# A Slice of the Cake
	desc = "chinese_factions.1.t"				# Now that we have achieved the unification of China, everybody want to rip the benefits. Already, three factions are fighting for power.
	picture = CMNG_COURT_eventPicture
	
	is_triggered_only = yes
	
	immediate = {
		add_faction = temples
		add_faction = enuchs
		add_faction = bureaucrats
		add_faction_influence = {
			faction = temples
			influence = 30
		}
		add_faction_influence = {
			faction = enuchs
			influence = 30
		}
		add_faction_influence = {
			faction = bureaucrats
			influence = 30
		}
		if = {
			limit = {
				NOT = {
					has_country_modifier = chinese_physiocratic_economy
					has_country_modifier = chinese_mercantilistic_economy
				}
			}
			country_event = { id = chinese_factions.2 days = 1 }
		}
	}
	
	option = {
		name = "chinese_factions.1.a"			# Favour the Temples
		add_faction_influence = {
			faction = temples
			influence = 10
		}
	}
	
	option = {
		name = "chinese_factions.1.b"			# Favour the Enuchs
		add_faction_influence = {
			faction = enuchs
			influence = 10
		}
	}
	
	option = {
		name = "chinese_factions.1.c"			# Favour the Bureaucrats
		add_faction_influence = {
			faction = bureaucrats
			influence = 10
		}
	}
}
